#include <iostream>
using namespace std;

class Car
{
    public:
    string make;
    string model;
    int year;
    //default constructor (no parameter)
    Car()
    {
        make = "BENZ";
        model = "xo8";
        year = 2;
    }
    void displayinfo()
    {
        cout<<year<<" "<<make<<" "<<model<<endl;
    }
};

int main()
{
    //creating anobject using default constuctor 
    Car myCar;

    //displaying the information
    myCar.displayinfo();

    return 0;

}
